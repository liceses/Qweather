### 1.2 AstronomyModel (~3h)

**文件**: `AstronomyModel.h`, `AstronomyModel.cpp`

**任务**:
1. 实现 `setSunTimes(sunrise, sunset)` — 解析 "HH:MM" → minutes 存储
2. 实现 `update(nowMsecs)`:
   - 计算 `nowMinutes` 从 QDateTime
   - 计算 `sunProgress = (nowMin - sunriseMin) / (sunsetMin - sunriseMin)`
   - 计算太阳高度角: `declination = 23.45° * sin(2π*(dayOfYear-81)/365)`, `hourAngle = (sunProgress-0.5)*π`, `altitude = asin(sin(lat)*sin(decl)+cos(lat)*cos(decl)*cos(hourAngle))`
   - 计算太阳方位角
   - 月亮位置: 与太阳对跖 `moonAltitude = sin((sunProgress+0.5)*π) * 0.7`
3. 实现 `sunProgress()` 访问器
4. 实现 `isNight()` / `isTwilight()` 判据

**接口**:
```cpp
class AstronomyModel : public QObject {
    Q_OBJECT
public:
    void setLocation(float lat, float lon);
    void setSunTimes(const QString &sunrise, const QString &sunset);  // "05:42"
    void setMoonData(int phaseIcon, float illumination);
    void update(qint64 nowMsecs);
    float sunProgress() const { return m_sunProgress; }
    bool isNight() const;      // sunProgress < -0.02 || > 1.02
    bool isTwilight() const;   // |sunProgress| < 0.05 || |sunProgress-1| < 0.05
    float solarAltitude() const { return m_solarAltitude; }
    // ...
};
```

**关键公式**:
```
sunProgress = (nowMin - sunriseMin) / (sunsetMin - sunriseMin)
// -0.5 凌晨2点, 0.0 日出, 0.5 正午, 1.0 日落, 1.5 晚间10点

declination = 23.45 * sin(2*PI*(dayOfYear - 81) / 365)  ← 度数
altitude = asin(sin(lat)*sin(descl) + cos(lat)*cos(decl)*cos(hourAngle))
azimuth = atan2(-cos(decl)*sin(hourAngle), sin(decl)*cos(lat)-cos(decl)*sin(lat)*cos(hourAngle))
```

**验证**: 输入 `lat=39.9, lon=116.4, sunrise="05:42", sunset="19:12"`, 正午 `sunProgress≈0.5`, `altitude≈75°`

**时序逻辑** (已在设计文档定义):
```
Cloud:  active: +100ms delay, 600ms fadeIn
         inactive: fadeOut 600ms → unload
Weather: active: +400ms(cloud) / +100ms delay, 600ms fadeIn
          inactive: fadeOut 600ms → unload
Fog:     active: +100ms delay, 400ms fadeIn
          inactive: fadeOut 400ms → unload
Lightning: instant (no transition)
```

**防抖实现**:
```cpp
void TransitionController::animateIn(float &tp, bool &active, int delayMs,
                                      void (TransitionController::*emitActive)(), 
                                      void (TransitionController::*emitTp)()) {
    int localId = m_transitionId;
    QTimer::singleShot(delayMs, this, [=, &tp, &active]() {
        if (m_transitionId != localId) return;
        active = true;     (this->*emitActive)();
        tp = 1.0f;         (this->*emitTp)();
    });
}
```

### 2.2 SkyLayer (天空 + 太阳 + 月亮 + 星星) (~4h)

**文件**: `shaders/sky.frag`, `SkyLayer.qml`

**shader 核心逻辑**:

```glsl
// sky.frag
#include "noise.glsl"    // hash, noise, fbm
#include "lighting.glsl" // sunGlow, moonGlow, starField

layout(std140, binding=0) uniform buf {
    mat4 qt_Matrix; float qt_Opacity;
    float time;
    float solarAltitude, solarAzimuth;
    float moonAltitude, moonAzimuth;
    float moonPhase, moonIllum;
    vec4 zenithColor, horizonColor, ambientColor;
    float exposure, twilightFactor;
    float starVisibility;
    float parallaxX, parallaxY;
} ubuf;

void main() {
    vec2 uv = qt_TexCoord0;
    
    // 1. 天空渐变
    vec3 sky = mix(ubuf.ambientColor.rgb, ubuf.zenithColor.rgb, uv.y);
    vec3 horizon = mix(sky, ubuf.horizonColor.rgb, smoothstep(0.3, 0.7, 1.0 - uv.y));
    
    // 2. 太阳光晕
    vec2 sunPos = vec2(
        0.5 + sin((ubuf.solarAzimuth - 180.0) * PI/180.0) * 0.5,
        0.7 - ubuf.solarAltitude / 90.0 * 0.6
    );
    float glow = sunGlow(uv, sunPos, 0.08 + (1.0 - ubuf.solarAltitude/90.0) * 0.12);
    vec3 sunColor = mix(vec3(1.0, 0.4, 0.2), vec3(1.0, 1.0, 0.95), sqrt(ubuf.solarAltitude/90.0));
    sky += sunColor * glow * ubuf.exposure;
    
    // 3. 月亮 (仅夜晚)
    if (ubuf.starVisibility > 0.01) {
        vec2 moonPos = vec2(
            0.5 + sin((ubuf.moonAzimuth - 180.0) * PI/180.0) * 0.45,
            0.75 - ubuf.moonAltitude / 90.0 * 0.45
        );
        sky += vec3(0.9, 0.88, 0.85) * moonGlow(uv, moonPos, ubuf.moonPhase) * ubuf.moonIllum * ubuf.starVisibility;
        sky += starField(uv, ubuf.time, ubuf.starVisibility);
    }
    
    fragColor = vec4(sky, 1.0) * ubuf.qt_Opacity;
}
```

**QML 包装**:
```qml

    property real time: globalClock.elapsed
    property real solarAltitude: skyState.solarAltitude
    // ... 所有 uniform 绑定 skyState
    fragmentShader: "qrc:/shaders/sky.frag.qsb"
    // fallback Rectangle (status !== Compiled)
}
```

### 2.4 CloudLayer (云噪声) (~3h)

**文件**: `shaders/cloud.frag`, `CloudLayer.qml`

**shader 核心**:
```glsl
// cloud.frag
#include "noise.glsl"

layout(...) uniform buf {
    mat4 qt_Matrix; float qt_Opacity;
    float time;
    float cloudCoverage, cloudOpacity;
    int variant;
    float windSpeed;
    float transitionProgress;
} ubuf;

void main() {
    float tp = ubuf.transitionProgress;
    float speed = 0.02 * ubuf.windSpeed;
    
    // 3 层 fBM 云噪声 (不同 speed)
    float c1 = fbm(vec2(qt_TexCoord0.x * 3.0 + ubuf.time * speed * 1.0, 
                         qt_TexCoord0.y * 4.0), ubuf.octaves);
    float c2 = fbm(vec2(qt_TexCoord0.x * 5.0 + ubuf.time * speed * 1.5, 
                         qt_TexCoord0.y * 6.0 + 2.0), ubuf.octaves);
    float cloud = c1 * 0.6 + c2 * 0.4;
    
    // tp 驱动: 云从右侧屏幕外漂入 + 覆盖率增加
    float cloudX = mix(0.6, 0.0, tp);  // tp=0: 云在+0.6(屏幕外), tp=1: 正常位置
    cloud = noise2D(vec2(qt_TexCoord0.x + cloudX, qt_TexCoord0.y) * 3.0);
    
    float coverage = mix(0.0, ubuf.cloudCoverage, tp);
    float mask = smoothstep(1.0 - coverage, 1.0 - coverage + 0.15, cloud);
    
    vec3 cloudColor = vec3(0.82, 0.84, 0.87);
    fragColor = vec4(cloudColor, mask * ubuf.cloudOpacity) * ubuf.qt_Opacity;
}
```

**验证**: CloudLayer 独立 → 白色云朵从右漂入 + 覆盖渐变

## 风险与应对

| 风险 | 可能性 | 应对 |
|------|:---:|------|
| GLSL shader Qt6 RHI 编译失败 (跨平台) | 中 | 每个 shader 先在一个平台验证通过; fallback Rectangle 已内置 |
| AstronomyModel 公式精确度不足 | 低 | 简化模型已够用; 如需精度可替换为 suncalc 算法 |
| WeatherProfile 68 码映射有遗漏 | 低 | 阶段 1 完成后跑全量 68 码映射 coverage 测试 |
| 过渡时序抖动 (Timer 不准) | 低 | `QTimer::singleShot` 在桌面端精度 <5ms, 足够 |
| perfLevel=低 仍不足 30fps (集显/4K) | 中 | 粒子极限降至 10, 云 noise 降至 1 octave |
| `qt_add_shaders` GLSL 版本不兼容 | 低 | 默认 `100es,120,150` 覆盖宽; 如需 440 级功能改为 `300es,330` |
